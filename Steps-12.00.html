Creating DAX Measures 12

Detailed Instructions:
 
1.    In Power BI Desktop, click on Final Booking table in the Data Pane.
2.    In the Home ribbon, click on New measure. 
3.    Type the following DAX codes in the formula bar one at a time (all the formulas are at the bottom) 
4.    After you done click on the green right mark so it will be saved
5.    Repeat the steps until all the measures are created.

 
. Total Passenger (Pax)= COUNTROWS ('Final Booking')
. Total Revenue = SUM ('Final Booking'[TotalFare_x])
. Avg Fare = AVERAGE ('Final Booking'[ TotalFare_x])
. Savings Rate = SUM ('Final Booking'[Savings_x]), SUM ('Final Booking'[TotalFare_x]), 0) 
  
.Passenger Wow = 
VAR CurrentWeek = WEEK NUM (MAX ('Calendar Table'[Date]))
VAR CurrentWeek Passenger= [. Pax]
VAR Prev Week = CurrentWeek - 1
VAR Prev Week Passenger = 
    CALCULATE (
        [Pax],
        FILTER (
            ALL ('Calendar Table'),
            WEEKNUM ('Calendar Table'[Date]) = PrevWeek
        )
    )
RETURN
IF(
    ISBLANK(PrevWeekPassenger),
    BLANK(),
    DIVIDE(CurrentWeekPassenger- PrevWeekPassenger, PrevWeekPassenger
)/100
)
 
 
.Revenue_WoW = 
VAR CurrentWeek = WEEKNUM(MAX('Calendar Table'[Date]))
VAR CurrentWeekRevenue = [.Total Revenue]
VAR PrevWeek = CurrentWeek - 1
VAR PrevWeekRevenue =

        [.Total Revenue],
        FILTER(
            ALL('Calendar Table'),
            WEEKNUM('Calendar Table'[Date]) = PrevWeek
        )
    )
RETURN
IF(
    ISBLANK(PrevWeekRevenue),
    BLANK(),
    DIVIDE(CurrentWeekRevenue - PrevWeekRevenue, PrevWeekRevenue)

.AvgFare_WoW = 
VAR CurrentWeek = WEEKNUM(MAX('Calendar Table'[Date]))
VAR CurrentWeekAvgPrice = [.Avg Fare]
VAR PrevWeek = CurrentWeek - 1
VAR PrevWeekAvgFare =

        [.Avg Fare],
        FILTER(
            ALL('Calendar Table'),
            WEEKNUM('Calendar Table'[Date]) = PrevWeek
        )
    )
RETURN
IF(
    ISBLANK(PrevWeekAvgFare),
    BLANK(),
    DIVIDE(CurrentWeekAvgPrice - PrevWeekAvgFare, PrevWeekAvgFare)
)/100
 
 .Savings_Rate_WoW =

VAR PrevWeek = CurrentWeek - 1

            ALL('Calendar Table'),
